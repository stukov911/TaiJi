# TaiJi
# program for drawing picture of Taiji 
import turtle as t
t.begin_fill()
t.color('black')
t.circle(200,180)
t.end_fill()
t.circle(200,180)
t.begin_fill()
t.color('white')
t.circle(100,180)
t.end_fill()
t.begin_fill()
t.color('black')
t.circle(-100,180)
t.end_fill()
t.penup()
t.goto(0,280)
t.pendown()
t.begin_fill()
t.color('white')
t.circle(20)
t.end_fill()
t.penup()
t.goto(0,80)
t.pendown()
t.begin_fill()
t.color('black')
t.circle(20)
t.end_fill()
t.hideturtle()
t.done()
